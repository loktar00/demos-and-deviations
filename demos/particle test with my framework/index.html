Clicked Effect

var emitter = new Emitter({pos : {x: this.cX, y: this.cY}});

// Initial Fire
emitter.addGroup({
	size: 10, 
	endSize: 1, 
	thrust : 0,
	thrustRange: {min:1, max:20}, 
	angle : 0,
	angleRange : {min:0, max:360}, 
	endAlpha: 0, 
	startColor : {r: 255, g:100, b:0}, 
	endColor: {r:255, g:255, b:0},
	duration : 1000,
	rate: 100,
	lifeTime : 500

});
